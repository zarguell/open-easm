```python
"""EPSS (Exploit Prediction Scoring System) integration.

Fetches daily EPSS scores from FIRST.org and caches them in cve_cache.
API docs: https://www.first.org/epss/api
"""
from __future__ import annotations

import logging
from typing import Any

import httpx

logger = logging.getLogger(__name__)

EPSS_API_URL = "https://api.first.org/data/v1/epss"
EPSS_BULK_URL = "https://epss.cyentia.com/epss_scores-current.csv.gz"
# The bulk URL returns a gzipped CSV with all CVE EPSS scores.
# Format: cve,epoch,date,score,percentile
# This is more efficient than per-CVE API calls.


async def refresh_epss_cache(pool: Any) -> int:
    """Download bulk EPSS scores and upsert into cve_cache.
    
    Returns the number of CVEs updated.
    """
    import gzip
    import csv
    import io
    
    upserted = 0
    async with httpx.AsyncClient(timeout=120.0, follow_redirects=True) as client:
        resp = await client.get(EPSS_BULK_URL)
        resp.raise_for_status()
    
    decompressed = gzip.decompress(resp.content).decode("utf-8")
    reader = csv.reader(io.StringIO(decompressed))
    
    # Skip header lines (start with #)
    rows = []
    for row in reader:
        if not row or row[0].startswith("#"):
            continue
        # Format: cve,epoch,date,score,percentile
        if len(row) >= 5:
            try:
                cve_id = row[0].strip()
                score = float(row[3].strip())
                percentile = float(row[4].strip())
                rows.append((cve_id, score, percentile))
            except (ValueError, IndexError):
                continue
    
    # Batch upsert
    for cve_id, score, percentile in rows:
        await pool.execute("""
            INSERT INTO cve_cache (cve_id, epss_score, epss_percentile, last_refreshed)
            VALUES ($1, $2, $3, now())
            ON CONFLICT (cve_id) DO UPDATE SET
                epss_score = EXCLUDED.epss_score,
                epss_percentile = EXCLUDED.epss_percentile,
                last_refreshed = now()
        """, cve_id, score, percentile)
        upserted += 1
    
    logger.info("epss cache refreshed", extra={"upserted": upserted})
    return upserted


async def lookup_epss(pool: Any, cve_id: str) -> dict[str, Any] | None:
    """Get EPSS score for a specific CVE."""
    row = await pool.fetchrow(
        "SELECT epss_score, epss_percentile FROM cve_cache WHERE cve_id = $1",
        cve_id,
    )
    if row is None or row["epss_score"] is None:
        return None
    return {"score": row["epss_score"], "percentile": row["epss_percentile"]}
```

```python
from datetime import UTC, datetime, timedelta

def compute_lifecycle_state(first_seen_at: datetime | None, now: datetime | None = None) -> str:
    """Classify entity lifecycle state based on first_seen_at."""
    if first_seen_at is None:
        return "unknown"
    now = now or datetime.now(UTC)
    age = now - first_seen_at
    if age < timedelta(hours=24):
        return "new"
    if age < timedelta(days=7):
        return "recent"
    return "stable"
```

```python
from __future__ import annotations
import logging
from typing import Any

from easm.config import NotificationChannel, NotificationConfig
from easm.notifications.types import NotificationPayload

logger = logging.getLogger(__name__)

# Severity ordering for min_severity filtering
_SEVERITY_ORDER = {"critical": 4, "high": 3, "medium": 2, "low": 1, "info": 0}

class NotificationDispatcher:
    """Dispatches notifications to configured channels based on severity rules."""
    
    def __init__(self, config: NotificationConfig, base_url: str = "http://localhost:8000"):
        self._config = config
        self._base_url = base_url
        self._rate_counts: dict[str, int] = {}  # channel_name -> count this hour
    
    async def dispatch(self, payload: NotificationPayload) -> None:
        """Send notification to all matching channels."""
        risk_level = _SEVERITY_ORDER.get(payload.risk, 0)
        
        for channel in self._config.channels:
            if not channel.enabled:
                continue
            
            min_level = _SEVERITY_ORDER.get(channel.min_severity, 0)
            if risk_level < min_level:
                continue
            
            # Rate limiting
            if self._is_rate_limited(channel.name):
                logger.warning("notification rate limited", extra={"channel": channel.name})
                continue
            
            payload.dashboard_url = f"{self._base_url}/ui/findings"
            
            try:
                handler = self._get_handler(channel)
                if handler:
                    await handler(channel, payload)
                    self._rate_counts[channel.name] = self._rate_counts.get(channel.name, 0) + 1
            except Exception:
                logger.exception("notification dispatch failed", extra={"channel": channel.name})
    
    def _is_rate_limited(self, channel_name: str) -> bool:
        count = self._rate_counts.get(channel_name, 0)
        return count >= self._config.rate_limit_per_hour
    
    def _get_handler(self, channel: NotificationChannel):
        if channel.type == "webhook":
            from easm.notifications.webhook import send_webhook
            return send_webhook
        elif channel.type == "slack":
            from easm.notifications.webhook import send_slack
            return send_slack
        elif channel.type == "smtp":
            from easm.notifications.smtp import send_email
            return send_email
        elif channel.type == "pagerduty":
            from easm.notifications.pagerduty import send_pagerduty
            return send_pagerduty
        return None
    
    def reset_hourly_counts(self) -> None:
        """Called hourly to reset rate limit counters."""
        self._rate_counts.clear()
```

```python
from __future__ import annotations
import hashlib
import hmac
import json
import logging
from typing import Any

import httpx

from easm.config import NotificationChannel
from easm.notifications.types import NotificationPayload

logger = logging.getLogger(__name__)


async def send_webhook(channel: NotificationChannel, payload: NotificationPayload) -> None:
    """Send notification to a generic webhook endpoint."""
    assert channel.webhook is not None
    cfg = channel.webhook
    
    body = json.dumps({
        "finding_id": payload.finding_id,
        "rule_id": payload.rule_id,
        "headline": payload.headline,
        "risk": payload.risk,
        "target_id": payload.target_id,
        "entity_ids": payload.entity_ids,
        "evidence": payload.evidence,
        "dashboard_url": payload.dashboard_url,
        "timestamp": payload.timestamp,
    })
    
    headers: dict[str, str] = {"Content-Type": "application/json"}
    if cfg.secret:
        signature = hmac.new(cfg.secret.encode(), body.encode(), hashlib.sha256).hexdigest()
        headers["X-EASM-Signature"] = f"sha256={signature}"
    
    async with httpx.AsyncClient(timeout=15.0) as client:
        resp = await client.request(cfg.method, cfg.url, content=body, headers=headers)
        resp.raise_for_status()
    
    logger.info("webhook sent", extra={"channel": channel.name, "url": cfg.url, "status": resp.status_code})


async def send_slack(channel: NotificationChannel, payload: NotificationPayload) -> None:
    """Send notification to Slack via webhook."""
    assert channel.slack is not None
    cfg = channel.slack
    
    risk_emoji = {"critical": "🔴", "high": "🟠", "medium": "🟡", "low": "🔵"}.get(payload.risk, "⚪")
    
    slack_body = {
        "text": f"{risk_emoji} *{payload.headline}*",
        "blocks": [
            {
                "type": "header",
                "text": {"type": "plain_text", "text": f"{risk_emoji} {payload.risk.upper()}: {payload.headline}"},
            },
            {
                "type": "section",
                "fields": [
                    {"type": "mrkdwn", "text": f"*Rule:* `{payload.rule_id}`"},
                    {"type": "mrkdwn", "text": f"*Risk:* {payload.risk.upper()}"},
                    {"type": "mrkdwn", "text": f"*Target:* `{payload.target_id}`"},
                    {"type": "mrkdwn", "text": f"*Time:* {payload.timestamp}"},
                ],
            },
            {
                "type": "actions",
                "elements": [
                    {
                        "type": "button",
                        "text": {"type": "plain_text", "text": "View in Dashboard"},
                        "url": payload.dashboard_url,
                    },
                ],
            },
        ],
    }
    
    async with httpx.AsyncClient(timeout=15.0) as client:
        resp = await client.post(cfg.url, json=slack_body)
        resp.raise_for_status()
    
    logger.info("slack notification sent", extra={"channel": channel.name})
```

```python
Time: {payload.timestamp}

View: {payload.dashboard_url}
"""


async def send_email(channel: NotificationChannel, payload: NotificationPayload) -> None:
    """Send notification via SMTP email."""
    assert channel.smtp is not None
    cfg = channel.smtp
    
    msg = MIMEMultipart("alternative")
    msg["Subject"] = f"[EASM {payload.risk.upper()}] {payload.headline}"
    msg["From"] = cfg.from_address
    msg["To"] = ", ".join(cfg.to_addresses)
    
    msg.attach(MIMEText(_build_email_text(payload), "plain"))
    msg.attach(MIMEText(_build_email_html(payload), "html"))
    
    try:
        if cfg.use_tls:
            server = smtplib.SMTP(cfg.host, cfg.port)
            server.starttls()
        else:
            server = smtplib.SMTP(cfg.host, cfg.port)
        
        if cfg.username and cfg.password:
            server.login(cfg.username, cfg.password)
        
        server.sendmail(cfg.from_address, cfg.to_addresses, msg.as_string())
        server.quit()
    except Exception:
        logger.exception("smtp send failed", extra={"channel": channel.name, "host": cfg.host})
        raise
    
    logger.info("email sent", extra={
        "channel": channel.name,
        "to": cfg.to_addresses,
        "subject": msg["Subject"],
    })
```

Note: Uses stdlib `smtplib` — no new dependencies. The handler runs `smtplib.SMTP` (blocking) inside an async function. For production, this should use `asyncio.get_event_loop().run_in_executor()` to avoid blocking the event loop. Wrap the SMTP block:

```python
import asyncio
# ...
loop = asyncio.get_event_loop()
await loop.run_in_executor(None, _send_sync, cfg, msg)
```
